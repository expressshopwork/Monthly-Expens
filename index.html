const SHEET_ID = '1gMal3LlP5Wsh9UqXZjtWg-5hF5bRdjuezrXUBsL_K_A';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);

    if (data.action === 'sync') {
      // Write transactions
      let sheet = ss.getSheetByName('Transactions') || ss.insertSheet('Transactions');
      sheet.clearContents();
      sheet.appendRow(['Date','Description','Amount','Type','Member','Category']);
      (data.transactions || []).forEach(t => {
        sheet.appendRow([t.date, t.desc, t.amount, t.type, t.member, t.category]);
      });

      // Write budget targets
      let bSheet = ss.getSheetByName('BudgetTargets') || ss.insertSheet('BudgetTargets');
      bSheet.clearContents();
      bSheet.appendRow(['Category','Target']);
      Object.entries(data.budgetTargets || {}).forEach(([cat, val]) => {
        bSheet.appendRow([cat, val]);
      });

      return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', message: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName('Transactions');
    const bSheet = ss.getSheetByName('BudgetTargets');
    const transactions = [];
    if (sheet && sheet.getLastRow() > 1) {
      const rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 6).getValues();
      rows.forEach(r => {
        transactions.push({ date:r[0], desc:r[1], amount:Number(r[2]), type:r[3], member:r[4], category:r[5] });
      });
    }
    const budgetTargets = {};
    if (bSheet && bSheet.getLastRow() > 1) {
      const bRows = bSheet.getRange(2, 1, bSheet.getLastRow()-1, 2).getValues();
      bRows.forEach(r => { if (r[0]) budgetTargets[r[0]] = Number(r[1]); });
    }
    return ContentService.createTextOutput(JSON.stringify({status:'ok', transactions, budgetTargets}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', message: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
